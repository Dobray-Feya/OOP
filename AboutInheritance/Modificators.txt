public class A
{
private int x; // доступен только внутри класса A
public int y; // доступен везде
internal int z; // доступен только в сборке (если модификатор не указан, то он internal)
protected int w; // доступен классу и всем наследникам
protected internal v; // доступен в классе, наследниках и всем из сборки
private protected u; // доступен в классе и наследниках только из этой же сборки
}
